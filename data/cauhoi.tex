\begin{ex}
Tập nghiệm của phương trình $x^{2} + 3 x + 1=0$ là
\choice
{$\left\{- \frac{3}{2} - \frac{\sqrt{5}}{2}, - \frac{3}{2} + \frac{\sqrt{5}}{2}\right\}$}
{$\left\{- \frac{3}{2} - \frac{\sqrt{5}}{2}, - \frac{3}{2} + \frac{\sqrt{5}}{2}\right\}$}
{$\left\{- \frac{3}{2} - \frac{\sqrt{5}}{2}, - \frac{3}{2} + \frac{\sqrt{5}}{2}\right\}$}
{$\emptyset$}
\loigiai{
Lời giải chi tiết.
}
\end{ex}
